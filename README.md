from machine import Pin 
import utime 

trigger = Pin(3, Pin.OUT)

def ultra(): 
    trigger.low() 
    utime.sleep_us(2)

    while echo.value() == 0:

        signalon = utime.ticks_us() 

    timepassed = signalon - signaloff 
    distance = (timepassed * 0.0343) / 2 
    print("La Distancia del Objeto es ",distance,"cm") 

while True: 
    ultra()
